-------------->>>> To Load a custom resource of type TEXT containing rtf data in a RichTextBox

	// Make sure RichText box library is loaded when creating a ATL dialog with RichText control
	// use the following in dialog's contructor
	// Explicitly load rich edit control
	// http://msdn.microsoft.com/en-us/library/dsezt3x7.aspx
	// LoadLibrary(_T("riched20.dll"));

	char buffer[4096];
		
	HRSRC hResource = FindResource(_AtlBaseModule.m_hInst, MAKEINTRESOURCE(IDR_ABOUTTEXT), L"TEXT");
	if (hResource)
	{
		HGLOBAL hLoadedResource = LoadResource(_AtlBaseModule.m_hInst, hResource);

		if (hLoadedResource)
		{
			LPVOID pLockedResource = LockResource(hLoadedResource);

			if (pLockedResource)
			{
				DWORD dwResourceSize = SizeofResource(_AtlBaseModule.m_hInst, hResource);

				if (0 != dwResourceSize)
				{
					memcpy_s(buffer, 4096, pLockedResource, dwResourceSize);
					buffer[dwResourceSize] = 0;	// Null terminated

					HWND hRichText = ::GetDlgItem(m_hWnd, IDC_ABOUTTEXT);
					::SendMessage(hRichText, WM_SETTEXT, 0, (LPARAM)buffer);
				}
			}
		}
	}
